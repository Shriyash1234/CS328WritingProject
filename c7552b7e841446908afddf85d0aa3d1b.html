import pandas as pd
import plotly.express as px
import plotly.graph_objs as go
from plotly.subplots import make_subplots

# Read in the CSV file
df = pd.read_csv('tax-reduction-in-income-inequality.csv')
df = df[(df['Year'] >= 2000) & (df['Year'] <= 2014)]

# Pivot the data so that each country has its own column
pivoted_df = df.pivot(index='Year', columns='Entity', values='Percentage reduction in Gini (OECD (2019))')
pivoted_df = pivoted_df.interpolate()
# Specify the countries to include
countries_to_include = ['Canada', 'France', 'Germany', 'United Kingdom', 'United States','Japan','South Korea','New Zealand']
pivoted_df = pivoted_df[countries_to_include]

# Create the plotly figure
fig = make_subplots(rows=1, cols=1)

# Add traces for each country
for country in pivoted_df.columns:
    fig.add_trace(go.Scatter(x=pivoted_df.index, y=pivoted_df[country], name=country))

# Update layout
fig.update_layout(title='Tax Reduction & Income Inequality',
                  xaxis_title='Year',
                  yaxis_title='Percentage reduction in Gini coefficients')

# Show the plot
fig.show()

import pandas as pd
import geopandas as gpd

import ipywidgets as widgets
from IPython.display import display

# Load the data
df4 = pd.read_csv('economic-inequality-gini-index.csv')
# Load the world map shapefile
world_map = gpd.read_file(gpd.datasets.get_path('naturalearth_lowres'))

# Define a function to update the map based on the selected year
def update_map(year):
    # Filter the data for the selected year
    df4_year = df4[df4['Year'] == year]

    # Merge the data with the world map based on the country code
    merged_data = world_map.merge(df4_year, left_on='iso_a3', right_on='Code', how='left')

    # Create a heatmap of the data on the world map
    fig, ax = plt.subplots(figsize=(20,10))
    merged_data.plot(column='Gini coefficient', cmap='YlOrRd', legend=True,
                     missing_kwds={'color': 'white', 'hatch': '///', 'label': 'Data Not Available'},
                 ax=ax)

    ax.set_title('Gini Coefficient - {}'.format(year))
    plt.show()

# Create a slider widget for selecting the year
year_slider = widgets.IntSlider(min=1980, max=2018, step=1, value=2014)

# Create a button widget to update the map
update_button = widgets.Button(description='Compare Map')

# Define a function to handle the button click event
def on_update_button_click(button):
    update_map(year_slider.value)

# Register the button click event handler
update_button.on_click(on_update_button_click)

# Display the widgets
display(year_slider)
display(update_button)

# Initial map display
update_map(year_slider.value)
